grammar ::= disjunction dlist
dlist ::= disjunction dlist
       |  epsilon
disjunction ::= NON_TERM '->' rule rlist ';'
rlist ::= '|' rule rlist
       |  epsilon
rule ::= conjunct clist
clist ::= '&' conjunct clist
       |  epsilon
conjunct ::= neg symbol slist
neg ::= '~' | epsilon
slist ::= symbol slist
       |  epsilon
symbol ::= NON_TERM
        |  '"' LITERAL '"' | 'alpha' | 'num' | 'alnum'
        |  'epsilon'
