grammar ::= rule rlist
rlist ::= rule rlist
       |  epsilon
rule ::= NON_TERM "=" rhs
rhs ::= conjunction conlist
conlist ::= "|" conjunction conlist
         |  epsilon
conjunction ::= conjunct clist
clist ::= "&" conjunct clist
       |  epsilon
conjunct ::= pos | neg
neg ::= "~" pos
pos ::= term tlist
tlist ::= term tlist
       |  epsilon
term ::= NON_TERM | terminal
terminal ::= '"' STR_LIT '"'
