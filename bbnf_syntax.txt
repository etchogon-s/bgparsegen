grammar ::= disjunction dlist
dlist ::= disjunction dlist
       |  epsilon
disjunction ::= NON_TERM "->" rhs ";"
rhs ::= rule rlist
rlist ::= "|" rule rlist
       |  epsilon
rule ::= conjunct clist
clist ::= "&" conjunct clist
       |  epsilon
conjunct ::= pos | neg
neg ::= "~" pos
pos ::= term tlist
tlist ::= term tlist
       |  epsilon
term ::= NON_TERM | terminal
terminal ::= '"' STR_LIT '"'
