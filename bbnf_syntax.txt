grammar ::= disjunction dlist
dlist ::= disjunction dlist
       |  epsilon
disjunction ::= NON_TERM '->' rhs ';'
rhs ::= rule rlist
rlist ::= '|' rule rlist
       |  epsilon
rule ::= conjunct clist
clist ::= '&' conjunct clist
       |  epsilon
conjunct ::= neg symbol slist
neg ::= '~' | epsilon
slist ::= symbol slist
       |  epsilon
symbol ::= NON_TERM | '"' STR_LIT '"'
